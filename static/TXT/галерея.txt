Шаг 3: Создание карточек с помощью JavaScript
Теперь добавим JavaScript, чтобы динамически генерировать карточки и реализовать пагинацию. Создайте файл script.js и добавьте следующий код:

JavaScript
Копировать
const itemsPerPage = 6; // Количество карточек на странице
const totalItems = 30; // Общее количество карточек
let currentPage = 1;

function generateCards(page) {
    const gallery = document.getElementById('gallery');
    gallery.innerHTML = ''; // Очищаем галерею перед добавлением новых карточек

    const start = (page - 1) * itemsPerPage;
    const end = Math.min(start + itemsPerPage, totalItems);

    for (let i = start; i < end; i++) {
        const card = `
            <div class="col-md-4 mb-4">
                <div class="card">
                    <img src="https://via.placeholder.com/300" class="card-img-top" alt="Card image">
                    <div class="card-body">
                        <h5 class="card-title">Карточка ${i + 1}</h5>
                        <p class="card-text">Описание карточки ${i + 1}.</p>
                    </div>
                </div>
            </div>
        `;
        gallery.innerHTML += card;
    }
}

function generatePagination() {
    const pagination = document.getElementById('pagination');
    pagination.innerHTML = ''; // Очищаем пагинацию перед добавлением новых элементов

    const pageCount = Math.ceil(totalItems / itemsPerPage);

    for (let i = 1; i <= pageCount; i++) {
        const pageItem = `
            <li class="page-item ${i === currentPage ? 'active' : ''}">
                <a class="page-link" href="#" onclick="changePage(${i})">${i}</a>
            </li>
        `;
        pagination.innerHTML += pageItem;
    }
}

function changePage(page) {
    currentPage = page;
    generateCards(currentPage);
    generatePagination();
}

// Инициализация
generateCards(currentPage);
generatePagination();

        ];

        const galleryContainer = document.getElementById('gallery');
        const paginationContainer = document.getElementById('pagination');
        const itemsPerPage = 6;
        let currentPage = 1;

        function displayImages(page) {
            galleryContainer.innerHTML = ''; // Очищаем контейнер
            const startIndex = (page - 1) * itemsPerPage;
            const endIndex = startIndex + itemsPerPage;
            const currentImages = images.slice(startIndex, endIndex);

            currentImages.forEach(imageSrc => {
                const img = document.createElement('img');
                img.src = imageSrc;
                img.classList.add('gallery-item', 'img-thumbnail');
                galleryContainer.appendChild(img);
            });
        }

        function setupPagination() {
            paginationContainer.innerHTML = ''; // Очищаем пагинацию
            const pageCount = Math.ceil(images.length / itemsPerPage);

            for (let i = 1; i <= pageCount; i++) {
                const listItem = document.createElement('li');
                listItem.classList.add('page-item');
                if (i === currentPage) {
                    listItem.classList.add('active');
                }
                const link = document.createElement('a');
                link.classList.add('page-link');
                link.href = '#';
                link.textContent = i;
                link.addEventListener('click', (event) => {
                    event.preventDefault();
                    currentPage = i;
                    displayImages(currentPage);
                    updateActivePage();
                });
                listItem.appendChild(link);
                paginationContainer.appendChild(listItem);
            }
        }

        function updateActivePage() {
            const pageLinks = paginationContainer.querySelectorAll('.page-item');
            pageLinks.forEach(linkItem => {
                linkItem.classList.remove('active');
                const pageNumber = parseInt(linkItem.querySelector('.page-link').textContent);
                if (pageNumber === currentPage) {
                    linkItem.classList.add('active');
                }
            });
        }

        // Инициализация галереи при загрузке страницы
        displayImages(currentPage);
        setupPagination();
    </script>

Вот что делает этот код:

HTML структура:

Подключает CSS Bootstrap для базовой стилизации.
Создает контейнер div с id="gallery" для отображения изображений.
Создает навигацию nav с ul элементом id="pagination" для кнопок пагинации.
Добавлены базовые стили CSS для галереи.
JavaScript:

Массив images содержит ссылки на изображения (в данном случае, placeholder-изображения). Ты заменишь их на свои реальные ссылки.
Получает ссылки на контейнеры галереи и пагинации.
itemsPerPage определяет количество изображений на одной странице.
currentPage отслеживает текущую отображаемую страницу.
displayImages(page) функция:
Очищает содержимое контейнера галереи.
Вычисляет начальный и конечный индексы изображений для текущей страницы.
Извлекает нужный фрагмент массива images.
Для каждого изображения создает элемент <img>, устанавливает src, добавляет классы Bootstrap (gallery-item, img-thumbnail) и добавляет его в контейнер галереи.
setupPagination() функция:
Очищает содержимое контейнера пагинации.
Вычисляет общее количество страниц.
Создает элементы <li> (с классом page-item) и <a> (с классом page-link) для каждой страницы.
Добавляет номер страницы в ссылку.
Добавляет обработчик события click для каждой ссылки, который предотвращает переход по ссылке, обновляет currentPage, вызывает displayImages() для отображения соответствующей страницы и updateActivePage() для обновления активной кнопки пагинации.
Добавляет созданный элемент <li> в контейнер пагинации.
updateActivePage() функция:
Удаляет класс active у всех элементов пагинации.
Добавляет класс active к элементу пагинации, соответствующему текущей странице.
При загрузке страницы вызываются displayImages(currentPage) и setupPagination() для первоначального отображения первой страницы и настройки кнопок пагинации.
Как использовать:

Сохрани HTML код в файл (например, index.html).
Замени placeholder-ссылки в массиве images на ссылки на твои изображения.
Открой файл index.html в браузере.
Ты увидишь галерею с пагинацией, стилизованную с помощью Bootstrap. При нажатии на кнопки пагинации будут отображаться соответствующие наборы изображений.

ТЗ

Необходимо сделать html документ для вывода объектов в галерею
вывод на экран в виде галереи в три ряда по 6 карточек будстрап
для больших экранов с адаптацией для меньших экранов
все с применением будстрап 5.
Адаптация для больших эранов 6 карточек в ряду, с уменьшением 4 в ряду для средних,
2 в ряду для более меньших экранов и 1 в ряду для самых маленьких экранов.

Карточка объекта должна содержать: фото, под фото титлл, под тилом кнопка "Подробнее".

Внизу галереии должна быть пагинация. В пагинации должны использоваться
следующие кнопки << переключение на первую страницу галереии,
>> переключение на последнию страницу галереии,
< переключение на группу из пяти указателей страниц назад,
> переключение на группу из указателей из пяти страниц вперед.
Между этими кнопками должна быть группа указателей страниц из пяти страниц
в виде квадратиков.
